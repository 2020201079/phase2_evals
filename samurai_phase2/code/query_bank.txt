SELECT ENAME
FROM PROJ, ASG, EMP
WHERE ASG.ENO = EMP.ENO
AND ASG.PNO = PROJ.PNO
AND EMP.ENAME != "J. Doe"
AND PROJ.PNAME = "CAD/CAM"
AND  (ASG.DUR = 12 OR ASG.DUR = 24)

FROM PROJ, ASG, EMP
WHERE ASG.ENO = EMP.ENO
AND ASG.PNO = PROJ.PNO
AND EMP.ENAME != "J. Doe"
AND PROJ.PNAME = "CAD/CAM"
AND  (ASG.DUR = 12 OR ASG.DUR = 24)

SELECT Employees.LastName, Orders.OrderID
FROM Orders,Employees
WHER Orders.EmployeeID = Employees.EmployeeID
Employees.LastName = 'Davolio' OR Employees.LastName = 'Fuller'
GROUP BY Employees.LastName
HAVING Orders.OrderID > 25;
